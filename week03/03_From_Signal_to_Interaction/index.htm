// Example 1B: Using a for-loop to blink 5 LEDs in sequence

int ledPins[] = {2, 3, 4, 5, 6};  // LED pins
int ledCount = 5;

void setup() {
  // Set all LED pins as outputs
  for (int i = 0; i < ledCount; i++) {
    pinMode(ledPins[i], OUTPUT);
  }
}

void loop() {
  // Turn each LED on and off one after another
  for (int i = 0; i < ledCount; i++) {
    digitalWrite(ledPins[i], HIGH);
    delay(150);
    digitalWrite(ledPins[i], LOW);
  }
}

if-else statements

// Example 2: If-Else for decision making

int buttonPin = 7;
int ledPin = 8;
int buttonState = 0;

void setup() {
  pinMode(buttonPin, INPUT); 
  pinMode(ledPin, OUTPUT);
}

void loop() {
  buttonState = digitalRead(buttonPin);

  if (buttonState == HIGH) {   // Button pressed
    digitalWrite(ledPin, HIGH);
  } else {                     // Button not pressed
    digitalWrite(ledPin, LOW);
  }
}

Boolean Toggle

// Example 3: Toggling an LED manually without using boolean logic
// (written as if we didn't know about ledState = !ledState;)

int buttonPin = 2;
int ledPin = 3;
int ledState = false;            // 0/LOW means OFF, 1/HIGH means ON
int lastButtonState = true;    // remember the last button reading

void setup() {
  pinMode(buttonPin, INPUT);
  pinMode(ledPin, OUTPUT);
}

void loop() {
  int buttonState = digitalRead(buttonPin);

  // Check if button is pressed (transition from HIGH to LOW)
  if (buttonState == false && lastButtonState == true) {

    // If LED is currently off, turn it on
    if (ledState == false) {
      ledState = true;
      digitalWrite(ledPin, HIGH);
    }
    // Otherwise, if LED is on, turn it off
    else if (ledState == true) {
      ledState = false;
      digitalWrite(ledPin, LOW);
    }

    delay(200); // debounce delay
  }

  lastButtonState = buttonState;
}


// Example 3A: Using bool != bool to toggle state on button press

int buttonPin = 2;
int ledPin = 3;
bool ledState = false;
bool lastButtonState = HIGH;

void setup() {
  pinMode(buttonPin, INPUT);
  pinMode(ledPin, OUTPUT);
}

void loop() {
  bool buttonState = digitalRead(buttonPin);

  // Detect button press transition
  if (buttonState == LOW && lastButtonState == HIGH) {
    // Toggle LED state
    ledState = !ledState;       // This means: ledState = ledState != true;
    digitalWrite(ledPin, ledState);
    delay(200); // Debounce delay
  }

  lastButtonState = buttonState;
}

Debouncing

// Example 4A: Button Debouncing with external pull-down resistor
//
// Wiring:
// - One side of the button → pin 2
// - Other side of the button → 5V
// - 10kΩ resistor from pin 2 → GND
// - LED (with 220Ω resistor) → pin 3 → GND

int buttonPin = 2;
int ledPin = 3;

bool ledState = false;          // current LED state
bool lastButtonState = LOW;     // last stable reading
unsigned long lastDebounceTime = 0;
unsigned long debounceDelay = 50;  // 50 ms debounce time window

void setup() {
  pinMode(buttonPin, INPUT);   
  pinMode(ledPin, OUTPUT);
}

void loop() {
  int reading = digitalRead(buttonPin);

  // If the button input changed, reset the debounce timer
  if (reading != lastButtonState) {
    lastDebounceTime = millis();
  }

  // Check if the input has been stable longer than debounceDelay
  if ((millis() - lastDebounceTime) > debounceDelay) {
    // If the stable state is HIGH, toggle the LED
    if (reading == HIGH) {
      ledState = !ledState;
      digitalWrite(ledPin, ledState);
    }
  }

  // Update lastButtonState for the next loop cycle
  lastButtonState = reading;
}

2.

From Sensor to Behavior

The basic loop:

sense → process → act

  • Sensor provides input (data)
  • Arduino interprets or maps it
  • Output gives feedback (light, motion, sound)

Example: Ultrasonic Sensor to LED

  • HC-SR04 ultrasonic distance sensor
  • LED connected via PWM (pin 9)

Concept: Distance controls LED brightness.

// Ultrasonic sensor to LED brightness
const int trigPin = 10;
const int echoPin = 11;
const int ledPin = 9;

void setup() {
  pinMode(trigPin, OUTPUT);
  pinMode(echoPin, INPUT);
  pinMode(ledPin, OUTPUT);
  Serial.begin(9600);
}

void loop() {
  digitalWrite(trigPin, LOW);
  delayMicroseconds(2);
  digitalWrite(trigPin, HIGH);
  delayMicroseconds(10);
  digitalWrite(trigPin, LOW);

  long duration = pulseIn(echoPin, HIGH);
  int distance = duration * 0.034 / 2;

  int brightness = map(distance, 2, 50, 255, 0);
  brightness = constrain(brightness, 0, 255);
  analogWrite(ledPin, brightness);

  Serial.println(distance);  // send to computer
  delay(50);
}

What is Serial Communication?

void setup(){ Serial.begin(9600) }

What is Serial Communication?

Method of sending data bit-by-bit over a single wire

Typical Use Cases

  • Real-time sensor input to visual/audio software
  • Physical interaction for design prototypes
  • Sound-responsive installations and spatial interfaces

Key Concept #1

Serial Basics & Setup

Use

Serial.begin(baudRate)

in

void setup()

Attention:

Only one program can use a serial port at the same time!

Key Concept #2

Data Formatting

Serial.print(data)

≠

Serial.println(data)

Formatting data:

to get:

val1,val2,val3

(comma-separated, newline-terminated)

Serial.print(val1) Serial.print(",") Serial.print(val2); Serial.print(","); Serial.println(val3);

Key Concept #3

Sensor Data Smoothing

Sensors are noisy → results fluctuate

Apply filters on Arduino before sending:

  • Moving Average
  • Exponential Smoothing
  • Median Filter

Trade-off:

More smoothing = less jitter, but slower response

Key Concept #4

Flow Control & Synchronization

  • Only send data when needed
  • Use Serial.available() on the Arduino side
  • Wait for full messages on the PC side

Advanced Tip:

Implement acknowledgements or handshakes for reliability in more complex systems.

Key Concept #5

Debugging Tools

Use Arduino Serial Monitor for testing

Print debug data separately from the serial data stream you use for your application

Best Practice:
Using Serial.print();

Isolate debugging output from structured communication

for example

if(debug) Serial.println(data);

Key Concept #6

Latency & Performance

Latency & Performance

  • Higher baud = faster data transfer (e.g., 115200)
  • Avoid delays delay() in loop where possible
  • Send compact messages

Baud rate

Serial.begin(9600); = 9600 bits/s

Serial.begin(115200); = 115200 bits/s

Always test performance in context

Visual/audio apps may choke on too much data

Key Concept #7

Cross-Platform Port Access

Port names vary by system:

  • Windows: COM3
  • macOS/Linux: /dev/tty.usbmodem... or /dev/cu.usbmodem... + number

Best Practice:
Use tools like:

Serial.list() (Processing)

serial.tools.list_ports (Python)

to list all available ports if you are not sure

Example Applications

  • Processing: sensor-driven visuals
  • TouchDesigner: digital installations or visuals
  • Python: data logging

Summary of Best Practices

  • Match baud rate & isolate communication
  • Format data clearly with delimiters (e.g., comma).
  • Preprocess noisy data on Arduino
  • Use serial tools to debug and test
  • Be mindful of platform and latency issues

Serial structure:

Arduino → Serial → Processing / TouchDesigner / Max

6. Example:

Arduino to Processing

// Example for controlling one value via Serial communication from an Arduino 

import processing.serial.*;

Serial myPort;   // Serial object
int sensorValue; // Variable to store the sensor input 

void setup() {
  size(800, 800);
  printArray(Serial.list());
  myPort = new Serial(this, "/dev/tty.usbmodem142101", 9600);
  myPort.bufferUntil('\n');
}

void draw() {
  background(0);
  float mapped = map(sensorValue, 5, 100, 0, width); // additional mapping 
  ellipse(width/2, height/2, mapped, mapped);
}

void serialEvent(Serial myPort) {
  String inString = myPort.readStringUntil('\n');
  if (inString != null) {
    sensorValue = int(trim(inString));
  }
}
